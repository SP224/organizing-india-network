const searchBox = document.getElementById("search"); const Graph = ForceGraph()(document.getElementById("graph")) .width(window.innerWidth) .height(window.innerHeight) .nodeAutoColorBy("cluster") .nodeLabel(node => node.label) .onNodeClick(node => { if (node.url) window.open(node.url, "_blank"); }); fetch("graph-data.json") .then(res => res.json()) .then(data => { Graph.graphData(data); searchBox.addEventListener("input", () => { const q = searchBox.value.toLowerCase(); Graph.nodeVisibility(node => node.label.toLowerCase().includes(q) ); }); });
